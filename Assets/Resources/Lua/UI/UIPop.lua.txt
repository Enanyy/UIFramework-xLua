local M = {
    mName = "UIPop",
}

setmetatable(M, UIBase)

--整个生命周期只调用一次
function M:OnLoad()
--BINDING_CODE_BEGIN	self.mButtonClose = self.gameObject.transform:Find('Widget/@Button.mButtonClose'):GetComponent(typeof(Button))
	self.mbuttonMain = self.gameObject.transform:Find('Widget/@Button.mbuttonMain'):GetComponent(typeof(Button))
	self.mButtonNormal = self.gameObject.transform:Find('Widget/@Button.mButtonNormal'):GetComponent(typeof(Button))
	self.mButtonWidget = self.gameObject.transform:Find('Widget/@Button.mButtonWidget'):GetComponent(typeof(Button))
	self.mText = self.gameObject.transform:Find('Widget/@Text.mText'):GetComponent(typeof(Text))
--BINDING_CODE_END
    
    self.mButtonClose.onClick:AddListener(function() self:Close() end )
    self.mbuttonMain.onClick:AddListener(function() UIManager:CloseAllAndOpen("UIMain") end)
    self.mButtonNormal.onClick:AddListener(function() UIManager:Open("UINormal") end)
    self.mButtonWidget.onClick:AddListener(function() UIManager:Open("UIWidget") end)
end

--整个生命周期只调用一次
function M:OnUnload()
--UNBINDING_CODE_BEGIN	self.mButtonClose = nil
	self.mbuttonMain = nil
	self.mButtonNormal = nil
	self.mButtonWidget = nil
	self.mText = nil
--UNBINDING_CODE_END
end

return M
