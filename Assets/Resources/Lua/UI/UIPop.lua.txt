local M = {
    mName = "UIPop",
}

setmetatable(M, UIBase)

--整个生命周期只调用一次
function M:OnLoad()
--BINDING_CODE_BEGIN	self.mButtonClose = self:GetComponent(Button, 'Widget/@Button.mButtonClose')
	self.mbuttonMain = self:GetComponent(Button, 'Widget/@Button.mbuttonMain')
	self.mButtonNormal = self:GetComponent(Button, 'Widget/@Button.mButtonNormal')
	self.mButtonWidget = self:GetComponent(Button, 'Widget/@Button.mButtonWidget')
	self.mText = self:GetComponent(Text, 'Widget/@Text.mText')
--BINDING_CODE_END
    
    self.mButtonClose.onClick:AddListener(function() self:Close() end )
    self.mbuttonMain.onClick:AddListener(function() UIManager:CloseAllAndOpen("UIMain") end)
    self.mButtonNormal.onClick:AddListener(function() UIManager:Open("UINormal") end)
    self.mButtonWidget.onClick:AddListener(function() UIManager:Open("UIWidget") end)
end

--整个生命周期只调用一次
function M:OnUnload()
--UNBINDING_CODE_BEGIN	self.mButtonClose = nil
	self.mbuttonMain = nil
	self.mButtonNormal = nil
	self.mButtonWidget = nil
	self.mText = nil
--UNBINDING_CODE_END
end

return M
