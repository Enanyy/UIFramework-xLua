local M = {
    mName = "UIWidget",
    mType = UIType.Widget,
    mFixedOrder = 1000
}

setmetatable(M, UIBase)

--整个生命周期只调用一次
function M:OnLoad()
--BINDING_CODE_BEGIN	self.mClose = self.gameObject.transform:Find('UIWidget/@Button.mClose'):GetComponent(typeof(Button))
	self.mNormal = self.gameObject.transform:Find('UIWidget/@Button.mNormal'):GetComponent(typeof(Button))
	self.mPop = self.gameObject.transform:Find('UIWidget/@Button.mPop'):GetComponent(typeof(Button))
	self.mMain = self.gameObject.transform:Find('UIWidget/@Button.mMain'):GetComponent(typeof(Button))
	self.mToggle = self.gameObject.transform:Find('UIWidget/@Toggle.mToggle'):GetComponent(typeof(Toggle))
--BINDING_CODE_END

    self.mClose.onClick:AddListener(function() self:Close() end )
    self.mPop.onClick:AddListener(function() UIManager:Open(UIPop) end)
    self.mNormal.onClick:AddListener(function() UIManager:Open(UINormal) end)
end

--整个生命周期只调用一次
function M:OnUnload()
--UNBINDING_CODE_BEGIN	self.mClose = nil
	self.mNormal = nil
	self.mPop = nil
	self.mMain = nil
	self.mToggle = nil
--UNBINDING_CODE_END
end


return M
