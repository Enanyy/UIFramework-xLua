local M = {
    mName = "UIWidget",
    mType = UIType.Widget,
    mFixedOrder = 1000
}

setmetatable(M, UIBase)

--整个生命周期只调用一次
function M:OnLoad()
--BINDING_CODE_BEGIN	self.mClose = self:GetComponent(Button, 'UIWidget/@Button.mClose')
	self.mNormal = self:GetComponent(Button, 'UIWidget/@Button.mNormal')
	self.mPop = self:GetComponent(Button, 'UIWidget/@Button.mPop')
	self.mMain = self:GetComponent(Button, 'UIWidget/@Button.mMain')
	self.mToggle = self:GetComponent(Toggle, 'UIWidget/@Toggle.mToggle')
--BINDING_CODE_END

    self.mClose.onClick:AddListener(function() self:Close() end )
    self.mPop.onClick:AddListener(function() UIManager:Open(UIPop) end)
    self.mNormal.onClick:AddListener(function() UIManager:Open(UINormal) end)
end

--整个生命周期只调用一次
function M:OnUnload()
--UNBINDING_CODE_BEGIN	self.mClose = nil
	self.mNormal = nil
	self.mPop = nil
	self.mMain = nil
	self.mToggle = nil
--UNBINDING_CODE_END
end


return M
