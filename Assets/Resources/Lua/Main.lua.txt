--用到时require
setmetatable(_G, {
    __index = function(t,i)
        local ret = require(i)
        if ret then
            t[i] = ret
        end
        return ret
    end
})

function main( )
    --UIManager:Open("UIMain")
    local data = DTHero.Get(1)
    print(data.id)
    print(data.name)
    print(data.height)
    print(data.list)

    UIManager:Init()
    UIManager:SetUILoader(CS.MainLua.LoadUI)
    UIManager:Open(UIMain)
    UIManager:Open('UIFixed',"UIMain")
end

main()
